    difficulty: basic
    type: fraction

  - prompt: 2/5 + 1/10 を計算しましょう。
    answer: 1/2

    difficulty: basic
    type: fraction
    hints:

  - prompt: 5/6 - 1/3 を計算しましょう。

    difficulty: basic
    type: fraction

    difficulty: standard
    type: fraction

    type: fraction

    type: fraction
